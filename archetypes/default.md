# Заголовок страницы. Формируется автоматически из имени файла: 
# дефисы заменяются на пробелы, слова с заглавной буквы.
title: "{{ replace .Name "-" " " | title }}"
# Дата создания страницы. Hugo подставит автоматически.
date: "{{ .Date }}"
# Дата последнего изменения страницы. По умолчанию совпадает с датой создания.
lastmod: "{{ .Date }}"
# Черновик? false — страница публикуется, true — скрыта.
draft: false
# Язык страницы. Hugo автоматически подставит "uk" или "ru" в зависимости от папки.
lang: "{{ .Page.Language.Lang }}"
# Ключ перевода для связи мультиязычных версий одной страницы.
# Обычно используется имя файла, должно быть уникальным.
translationKey: "{{ .Name }}"
# Мета-описание страницы для SEO. Рекомендуется заполнять вручную при создании страницы.

# Ключевые слова для SEO. Заполняются вручную.
keywords: ["ключевое_слово1", "ключевое_слово2"]
# Slug (ЧПУ) — часть URL страницы. 
# | lower приводит все символы к нижнему регистру для SEO-дружелюбного URL.
slug: "{{ .Name | lower }}"
# Канонический URL для SEO. Заполняется вручную при необходимости.
canonicalURL: ""
# Показывать оглавление (Table of Contents). По умолчанию false.
showToc: false
# Показывать метаданные (автор, дата) на странице.
hidemeta: false
# Включить комментарии на странице. false — комментарии выключены.
comments: false
# Показывать хлебные крошки (навигация по сайту).
showBreadCrumbs: true
# Показывать ссылки на предыдущие/следующие страницы.
showPostNavLinks: true
# Параметры обложки страницы
cover:
  # Путь к изображению. Обычно размещается в /images/
  image: "/images/default-cover.webp"
  # Альтернативный текст для SEO и доступности (читатели с экранными читалками)
  alt: "Обложка страницы {{ .Name }}"
  # true — путь относительный, false — абсолютный
  relative: true
  # true — скрыта, false — видима
  hidden: false
  # Подпись под изображением (видима для читателя)
  caption: "Подпись к странице {{ .Name }}"
# JSON-LD Schema.org для поисковых систем
# Помогает поисковикам понимать страницу как объект WebPage.
schema:
  type: WebPage
  # Название страницы
  name: "{{ replace .Name "-" " " | title }}"
  # Описание страницы
  description: "Краткое описание страницы {{ .Name }} для поисковых систем"
  # Указывает, что именно эта страница является основной для данного объекта
  mainEntityOfPage: "{{ .Permalink }}"
# Настройки PWA (Progressive Web App)
pwa:
  # Путь к манифесту приложения
  manifest: "/manifest.webmanifest"
  # Страница для офлайн-режима
  offline: "/offline.html"
  # Цвет темы браузера (виден в мобильных браузерах)
  themeColor: "#0A0A0A"
  # Цвет фона splash screen при запуске PWA
  backgroundColor: "#FFFFFF"
# Кнопка "Редактировать пост"
editPost:
  # Ссылка на репозиторий с контентом (замени на свой репозиторий)

  # Текст кнопки (можно локализовать)
  text: "Запропонувати зміни"
  # true — автоматически добавлять путь к файлу к ссылке
  appendFilePath: true
