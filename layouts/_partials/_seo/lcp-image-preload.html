{{- /* Preload the page's primary visible image when the project convention makes it safe to infer that the front matter image is also the first content image. The matching seo-image shortcode skips its inline preload for these pages, so the browser discovers one responsive WebP preload from instead of a late duplicate next to the visible image element. */ -}} {{- $schemaTypes := .Params.schema_types | default (slice) -}} {{- $isPrimaryContentPage := and .IsPage (or (in $schemaTypes "article") (in $schemaTypes "news") (in $schemaTypes "product")) -}} {{- $isProductPage := and .IsPage (in $schemaTypes "product") -}} {{- $coverHiddenInSingle := (.Param "cover.hiddenInSingle") | default false -}} {{- $src := .Params.image | default "" -}} {{- $sizes := .Params.seo_image_sizes | default "(min-width: 1198px) 1150px, (max-width: 768px) calc(100vw - 28px), calc(100vw - 48px)" -}} {{- $isPaginatedIndexPage := .Scratch.Get "isPaginatedIndexPage" | default false -}} {{- if and .IsHome (not $isPaginatedIndexPage) -}} {{- with resources.Get "images/home-hero85.webp" -}} {{- $hero := . -}} {{- $heroSizes := "(min-width: 1280px) 50vw, (min-width: 1024px) 42vw, 100vw" -}} {{- $heroSrcset := slice -}} {{- range (slice 320 480 640 768 1024 1280 1600 2102) -}} {{- if le . $hero.Width -}} {{- $variant := $hero.Resize (printf "%dx webp q72" .) -}} {{- $heroSrcset = $heroSrcset | append (printf "%s %dw" $variant.RelPermalink $variant.Width) -}} {{- end -}} {{- end -}} {{- $heroPreload := $hero.Resize "768x webp q72" -}} {{- $heroSrcsetAttr := delimit $heroSrcset ", " -}} {{- end -}} {{- end -}} {{- if and $isPrimaryContentPage $coverHiddenInSingle $src -}} {{- with .Resources.GetMatch $src -}} {{- $img := . -}} {{- if and $isProductPage (reflect.IsImageResourceProcessable $img) -}} {{- $gallerySizes := "(min-width: 1024px) 55vw, calc(100vw - 2rem)" -}} {{- $fallbackWidth := 1200 -}} {{- if lt $img.Width $fallbackWidth -}} {{- $fallbackWidth = $img.Width -}} {{- end -}} {{- $webp := $img.Resize (printf "%dx webp q82" $fallbackWidth) -}} {{- $webpSrcset := slice -}} {{- range (slice 480 720 960 1200 1600) -}} {{- if le . $img.Width -}} {{- $variant := $img.Resize (printf "%dx webp q82" .) -}} {{- $webpSrcset = $webpSrcset | append (printf "%s %dw" $variant.RelPermalink $variant.Width) -}} {{- end -}} {{- end -}} {{- $fallbackSrcsetItem := printf "%s %dw" $webp.RelPermalink $webp.Width -}} {{- if not (in $webpSrcset $fallbackSrcsetItem) -}} {{- $webpSrcset = $webpSrcset | append $fallbackSrcsetItem -}} {{- end -}} {{- $webpSrcsetAttr := delimit $webpSrcset ", " -}} {{- else if reflect.IsImageResourceProcessable $img -}} {{- $webp := $img.Fill (printf "%dx%d webp q82" $img.Width $img.Height) -}} {{- $webpSrcset := slice -}} {{- $candidateWidths := slice 360 480 640 768 1024 1280 1536 1920 -}} {{- range $candidateWidths -}} {{- if lt . $img.Width -}} {{- $candidateHeight := div (mul . $img.Height) $img.Width -}} {{- $variant := $img.Fill (printf "%dx%d webp q82" . $candidateHeight) -}} {{- $webpSrcset = $webpSrcset | append (printf "%s %dw" $variant.RelPermalink $variant.Width) -}} {{- end -}} {{- end -}} {{- $webpSrcset = $webpSrcset | append (printf "%s %dw" $webp.RelPermalink $webp.Width) -}} {{- $webpSrcsetAttr := delimit $webpSrcset ", " -}} {{- else if eq $img.MediaType.SubType "svg" -}} {{- end -}} {{- end -}} {{- end -}}
